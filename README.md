               Detection Methods

      Flags transactions whose amount exceeds 2.5 standard
      deviations from the mean across all transactions.

  [2] Duplicate Detection
      Identifies rows sharing the same amount, merchant,

  [3] Large Round Amount
      Flags amounts >= Rs.10,000 that are rounded to the
      nearest Rs.1,000 (potential cash stuffing).

data_loader.py
--------------
Load and validate transaction CSV data.
"""

import pandas as pd
import os
import sys


REQUIRED_COLUMNS = {
    "transaction_id", "date", "amount", "type",
    "category", "merchant", "balance", "description"
}


def load_transactions(filepath: str) -> pd.DataFrame:
    """
    Load transaction data from a CSV file.

    Args:
        filepath: Path to the CSV file.

    Returns:
        pd.DataFrame: Raw transaction data.

    Raises:
        FileNotFoundError: If the CSV file does not exist.
        ValueError: If required columns are missing.
    """
    if not os.path.exists(filepath):
        raise FileNotFoundError(f"File not found: {filepath}")

    df = pd.read_csv(filepath)

    missing = REQUIRED_COLUMNS - set(df.columns)
    if missing:
        raise ValueError(f"Missing required columns: {missing}")

    print(f"[OK] Loaded {len(df):,} transactions from '{filepath}'")
    return df


def get_summary(df: pd.DataFrame) -> dict:
    """Return a high-level summary of the raw dataframe."""
    return {
        "total_transactions": len(df),
        "columns":            list(df.columns),
        "date_range":         (df["date"].min(), df["date"].max()),
        "null_counts":        df.isnull().sum().to_dict(),
        "duplicate_rows":     df.duplicated().sum(),
    }


if __name__ == "__main__":
    path = sys.argv[1] if len(sys.argv) > 1 else "data/transactions.csv"

"""
cleaner.py
----------
Clean and preprocess raw transaction data.
"""

import pandas as pd
import numpy as np


def clean_transactions(df: pd.DataFrame) -> pd.DataFrame:
    """
    Full cleaning pipeline for raw transaction data.

    Steps:
    1. Parse and sort dates
    2. Drop duplicate transaction IDs
    3. Fill / drop missing values
    4. Normalize string columns
    5. Ensure correct data types
    """
    df = df.copy()

    # 1. Parse dates
    df["date"] = pd.to_datetime(df["date"], errors="coerce")
    before = len(df)
    df = df.dropna(subset=["date"])
    if len(df) < before:
        print(f"[!] Dropped {before - len(df)} rows with unparseable dates.")

    df = df.sort_values("date").reset_index(drop=True)

    # 2. Remove duplicate transaction IDs
    dupes = df.duplicated(subset="transaction_id").sum()
    if dupes:
        print(f"[!] Removed {dupes} duplicate transaction IDs.")
        df = df.drop_duplicates(subset="transaction_id")

    # 3. Handle missing values
    df = df.dropna(subset=["amount"])
    df["description"] = df["description"].fillna(df["merchant"])

    # 4. Normalize string columns
    for col in ["type", "category", "merchant", "description"]:
        df[col] = df[col].astype(str).str.strip()

    df["type"]     = df["type"].str.upper()
    df["category"] = df["category"].str.title()

    # 5. Ensure numeric types
    df["amount"]  = pd.to_numeric(df["amount"],  errors="coerce").abs()
    df["balance"] = pd.to_numeric(df["balance"], errors="coerce")

    # Add derived columns
    df["month"]       = df["date"].dt.to_period("M").astype(str)
    df["day_of_week"] = df["date"].dt.day_name()
    df["week"]        = df["date"].dt.isocalendar().week.astype(int)

    print(f"[OK] Cleaned data: {len(df):,} transactions retained.")
    return df


def remove_outliers(df: pd.DataFrame,
                    column: str = "amount",
                    z_thresh: float = 4.0) -> pd.DataFrame:
    """Remove extreme statistical outliers from a numeric column."""
    from scipy.stats import zscore
    z = np.abs(zscore(df[column].dropna()))
    mask = z < z_thresh
    removed = (~mask).sum()
    print(f"[!] Removed {removed} outliers from '{column}' (|Z| > {z_thresh}).")
    return df[mask].reset_index(drop=True)

"""
analyzer.py
-----------
Core financial analysis functions for bank transactions.
"""

import pandas as pd
import numpy as np


def monthly_summary(df: pd.DataFrame) -> pd.DataFrame:
    """
    Summarize total debit/credit amounts and transaction counts
    per month.

    Returns:
        DataFrame with columns:
        month, total_debit, total_credit, net, txn_count
    """
    debits = (
        df[df["type"] == "DEBIT"]
        .groupby("month")["amount"]
        .agg(total_debit="sum", debit_count="count")
    )
    credits = (
        df[df["type"] == "CREDIT"]
        .groupby("month")["amount"]
        .agg(total_credit="sum", credit_count="count")
    )
    summary = debits.join(credits, how="outer").fillna(0)
    summary["net"]       = summary["total_credit"] - summary["total_debit"]
    summary["txn_count"] = summary["debit_count"] + summary["credit_count"]
    return summary.reset_index()


def category_breakdown(df: pd.DataFrame) -> pd.DataFrame:
    """Return spending breakdown by category (debits only)."""
    debits = df[df["type"] == "DEBIT"]
    cat = (
        debits.groupby("category")["amount"]
        .agg(total="sum", avg="mean", count="count")
        .sort_values("total", ascending=False)
        .reset_index()
    )
    cat["pct"] = (cat["total"] / cat["total"].sum() * 100).round(2)
    return cat


def top_merchants(df: pd.DataFrame, n: int = 10) -> pd.DataFrame:
    """Return top N merchants by total spend."""
    debits = df[df["type"] == "DEBIT"]
    return (
        debits.groupby("merchant")["amount"]
        .sum()
        .sort_values(ascending=False)
        .head(n)
        .reset_index()
        .rename(columns={"amount": "total_spent"})
    )


def daily_spend(df: pd.DataFrame) -> pd.DataFrame:
    """Return total daily spend."""
    return (
        df[df["type"] == "DEBIT"]
        .groupby("date")["amount"]
        .sum()
        .reset_index()
        .rename(columns={"amount": "daily_spend"})
    )


def weekday_analysis(df: pd.DataFrame) -> pd.DataFrame:
    """Return average spend per day of the week."""
    order = ["Monday", "Tuesday", "Wednesday", "Thursday",
             "Friday", "Saturday", "Sunday"]
    result = (
        df[df["type"] == "DEBIT"]
        .groupby("day_of_week")["amount"]
        .mean()
        .reindex(order)
        .reset_index()
        .rename(columns={"amount": "avg_spend"})
    )
    return result


def balance_over_time(df: pd.DataFrame) -> pd.DataFrame:
    """Return last known balance per day."""
    return (
        df.sort_values("date")
        .groupby("date")["balance"]
        .last()
        .reset_index()
    )


def compute_statistics(df: pd.DataFrame) -> dict:
    """Compute overall financial statistics."""
    debits  = df[df["type"] == "DEBIT"]["amount"]
    credits = df[df["type"] == "CREDIT"]["amount"]
    return {
        "total_debit":        round(debits.sum(), 2),
        "total_credit":       round(credits.sum(), 2),
        "net_flow":           round(credits.sum() - debits.sum(), 2),
        "avg_transaction":    round(df["amount"].mean(), 2),
        "max_transaction":    round(df["amount"].max(), 2),
        "min_transaction":    round(df["amount"].min(), 2),
        "std_transaction":    round(df["amount"].std(), 2),
        "total_transactions": len(df),
        "unique_merchants":   df["merchant"].nunique(),
        "unique_categories":  df["category"].nunique(),
    }

"""
anomaly_detector.py
-------------------
Detect suspicious/anomalous transactions using statistical methods.
"""

import pandas as pd
import numpy as np
from scipy.stats import zscore


def detect_zscore_anomalies(df: pd.DataFrame,
                             threshold: float = 2.5) -> pd.DataFrame:
    """
    Flag transactions whose amount Z-score exceeds the threshold.

    Args:
        df:        Cleaned transaction DataFrame.
        threshold: Z-score cutoff (default 2.5).

    Returns:
        DataFrame of flagged transactions with 'z_score' column added.
    """
    df = df.copy()
    df["z_score"] = np.abs(zscore(df["amount"].fillna(0))).round(3)
    anomalies = (
        df[df["z_score"] >= threshold]
        .sort_values("z_score", ascending=False)
    )
    print(f"[!] Detected {len(anomalies)} anomalous transactions "
          f"(Z >= {threshold}).")
    return anomalies.reset_index(drop=True)


def detect_duplicate_transactions(df: pd.DataFrame) -> pd.DataFrame:
    """
    Find potential duplicate transactions:
    same amount, merchant, and date.

    Returns:
        DataFrame of suspected duplicates.
    """
    dupes = df[
        df.duplicated(subset=["date", "amount", "merchant"], keep=False)
    ]
    print(f"[!] Found {len(dupes)} potential duplicate transactions.")
    return dupes.sort_values(["merchant", "date"]).reset_index(drop=True)


def detect_large_round_amounts(df: pd.DataFrame,
                                min_amount: float = 10000) -> pd.DataFrame:
    """
    Flag large round-number transactions
    (potential cash stuffing / suspicious activity).

    Args:
        df:         Cleaned transaction DataFrame.
        min_amount: Minimum amount to consider.

    Returns:
        DataFrame of flagged round-amount transactions.
    """
    mask = (df["amount"] >= min_amount) & (df["amount"] % 1000 == 0)
    flagged = df[mask].copy()
    flagged["flag_reason"] = "Large round amount"
    print(f"[!] Found {len(flagged)} large round-amount transactions.")
    return flagged.reset_index(drop=True)


def detect_late_night_transactions(df: pd.DataFrame) -> pd.DataFrame:
    """Flag transactions occurring between midnight and 4 AM."""
    if "hour" not in df.columns:
        df = df.copy()
        df["hour"] = pd.to_datetime(df["date"]).dt.hour

    late = df[df["hour"].between(0, 4)].copy()
    late["flag_reason"] = "Late-night transaction (00:00-04:00)"
    print(f"[!] Found {len(late)} late-night transactions.")
    return late.reset_index(drop=True)
